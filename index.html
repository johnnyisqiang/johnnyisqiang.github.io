""" # 本地接口：获取基金列表 @app.route("/api/get-funds") def api_get_funds(): codes = json.loads(open("funds.json", "r", encoding="utf-8").read()) result = [] for code in codes: result.append(fetch_fund_data(code)) return json.dumps(result, ensure_ascii=False) # 主页 @app.route("/") def index(): return render_template_string(HTML_TEMPLATE) # 运行本地服务器（端口10001） def run_server(): app.run(host="127.0.0.1", port=10001, debug=False) # 保存基金代码 def save_codes(codes): json.dump(codes, open("funds.json", "w", encoding="utf-8"), ensure_ascii=False) # 主程序 if __name__ == "__main__": # 初始化基金列表文件 try: open("funds.json", "r") except: save_codes(["017893", "005693", "013817"]) Thread(target=run_server, daemon=True).start() print("✅ 本地服务已启动 → http://127.0.0.1:10001") input("按回车退出\n")
